## Sample code

```cpp
#include "lis2mdl_class.h"
DevI2C *i2c;
LIS2MDL *lis2mdl;
int32_t axes[3];
int16_t raw[3];
uint8_t id;
void setup(){
    i2c = new DevI2C(D14, D15);
    lis2mdl = new LIS2MDL(*i2c);
    // init
    lis2mdl->init(NULL);
}
void loop(){
    // read id
    lis2mdl->readId(&id);
    Serial.printf("Id: %d\n", id);
    // getMAxes
    lis2mdl->getMAxes(axes);
    Serial.printf("Axes: x - %d, y - %d, z - %d\n", axes[0], axes[1], axes[2]);
    delay(1000);
}
```
